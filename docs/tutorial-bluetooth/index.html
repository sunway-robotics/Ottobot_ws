void loop() {
  if (BluetoothSerial.available()) {
    char command = BluetoothSerial.read();
    executeCommand(command);
  }
}

void executeCommand(char command) {
    switch (command) {
      case FORWARD:
        Otto.walk(1, 1000, 1);
        break;
      case BACKWARD:
        Otto.walk(1, 1000, -1);
        break;
      case LEFT:
        Otto.turn(1, 1000, 1);
        break;
      case RIGHT:
        Otto.turn(1, 1000, -1);
        break;
      case SQUARE:
        Otto.bend(1, 1000, 1);
        break;
      case CIRCLE:
        Otto.bend(1, 1000, -1);
        break;
      case TRIANGLE:
        Otto.moonwalker(1, 1000, -1);
        break;
      case START:
        Otto.shakeLeg(1, 1000, 1);
        break;
      case CROSS:
        Otto.home();
        break;
      default:
        break;
    }
  }
